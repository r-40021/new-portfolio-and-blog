{{- $file := "package.json" -}} {{- $basePath := .Site.Params.docs.basePath -}} {{- $file = path.Join $basePath (path.Clean $file) -}} {{- $regex := printf `"version":.*(\r\n|\r|\n)` -}} {{- $match := findRE $regex (readFile $file) -}} {{- $match = index $match 0 -}} {{- if $match -}} {{- printf "v%s" (trim (index (split $match ":") 1) " \r\n,\"") -}} {{- end -}}
